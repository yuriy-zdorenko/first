#include
                
using namespace std;
                
int main()
{
    SetConsoleCP(1251);
    SetConsoleOutputCP(1251);
                
    srand(time(NULL)); // встановлення початкової точки, з якої відбуватиметься генерування випадкових чисел
                
    char key; // змінна для виходу з циклу
    int user; // вибір гравця
    int computer; // вибір комп'ютера
                
    cout << "~~~~~~~~~~~~~~ Камінь-ножиці-папір ~~~~~~~~~~~~~~" << endl;
    while (true)
    {
        do
        {
            cout << "Ваш хід. Камінь - 1, ножиці - 2, папір - 3:\t";
            cin >> user;

            if (user < 1 || user > 3) // перевірка чи введене значення задовольняє умову
            {
                cout << "~~~~~~~ Помилка! Введіть число від 1 до 3 ~~~~~~~" << endl;
            }
        } while (user < 1 || user > 3);
                
        cout << "~~~~~~~~~~~~ Комп'ютер вибрав ";
        computer = 1 + rand() % 3; // випадкове число від 1 до 3
                
        switch (computer) // обробка вибору комп'ютера
        {
        case 1: cout << "камінь ~~~~~~~~~~~~" << endl;
            break;
        case 2: cout << "ножиці ~~~~~~~~~~~~" << endl;
            break;
        case 3: cout << "папір ~~~~~~~~~~~~~" << endl;
            break;
        default:
            break;
        }
                
        switch (user - computer) // визначаємо результат гри
        {
        case  0: cout << "~~~~~~~~~~~~~~~~~~~~ Нічия! ~~~~~~~~~~~~~~~~~~~~~" << endl;
            break;
        case -1:
        case  2: cout << "~~~~~~~~~~~~~~~~~ Ви перемогли! ~~~~~~~~~~~~~~~~~" << endl;
            break;
        case  1:
        case -2: cout << "~~~~~~~~~~~~~~ Переміг комп'ютер! ~~~~~~~~~~~~~~~" << endl;
        default:
            break;
        }
                
        cout << "Бажаєте зіграти ще раз? (Y/N):\t\t\t";
        cin >> key;
                
        if (key == 'N' || key == 'n')
            break;
        else
            cout << endl;
    }
    cout << "~~~~~~~~~~~~~~~~~ Гра завершена ~~~~~~~~~~~~~~~~~" << endl;
    return 0;
}
