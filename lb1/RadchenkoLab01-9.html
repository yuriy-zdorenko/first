Програмування

fn f(x: f64) -> f64 {
	//let result = x.powi(2) * (2.0 * x).cos() + 0.2;
	let result = 1.0 / x.tan() - 0.1;
	result
}

fn main() {
	let eps = 0.005;
	let mut a = 4.6;
	let b = 4.65;
	let mut x = a;
	let mut n = 0;
	loop {
		a = x;
		x = a - (f(a) * (b - a)) / (f(b) - f(a));
		let dx = (x - a).abs();
		println!("{0}. x{0} = {1:.6}; x{2} = {3:.6}; dx = {4:.6};", n, a, n+1, x, dx);
		n += 1;
		if dx <= eps { break; }
	}
	println!("Результат: {:.6}", x);
}
