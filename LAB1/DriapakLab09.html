from django.shortcuts import render, redirect, get_object_or_404
from django.shortcuts import render, redirect
from .models import ComplexNumber, ComplexArray
from .forms import ComplexNumberForm
from datetime import datetime
from django.shortcuts import render
from django.http import HttpRequest
from .models import ComplexArray
from .forms import OperationForm


def home(request):
    """Renders the home page."""
    assert isinstance(request, HttpRequest)
    return render(
        request,
        'app/complex_operations.html',
        {
            'title':'Home Page',
            'year':datetime.now().year,
        }
    )
def complex_numbers(request):
    result = None
    # Ініціалізація форм
    form = ComplexNumberForm()  # Ініціалізація форми ComplexNumberForm
    op_form = OperationForm()  # Ініціалізація форми OperationForm

    if request.method == "POST":
        if 'add_complex' in request.POST:
            # Якщо натиснуто кнопку додавання комплексного числа
            form = ComplexNumberForm(request.POST)
            if form.is_valid():
                # Збереження нового комплексного числа в базі даних
                form.save()
                return redirect('complex_numbers')
        elif 'operate' in request.POST:
            # Якщо натиснуто кнопку виконання операції
            op_form = OperationForm(request.POST)
            if op_form.is_valid():
                # Виконання обраної операції над комплексними числами
                operation = op_form.cleaned_data['operation']
                z1 = op_form.cleaned_data['z1'].complex
                z2 = op_form.cleaned_data['z2'].complex
                ca = ComplexArray([z1, z2])
                result = ca.operate(z1, z2, operation)

    # Отримання всіх комплексних чисел з бази даних
    numbers = ComplexNumber.objects.all()
    # Створення списку комплексних чисел
    complex_list = [n.complex for n in numbers]
    # Створення об'єкта ComplexArray з отриманого списку комплексних чисел
    ca = ComplexArray(complex_list)
    # Обчислення модулів сум сусідніх комплексних чисел
    modulus_sums = ca.modulus_sums()

    # Відображення шаблону complex_numbers.html з передачею відповідних даних
    return render(request, 'app/complex_numbers.html', {
        'form': form,
        'op_form': op_form,
        'numbers': numbers,
        'modulus_sums': modulus_sums,
        'result': result,
    })

def delete_number(request, number_id):
    # Видалення комплексного числа за його ідентифікатором
    number = get_object_or_404(ComplexNumber, pk=number_id)
    number.delete()
    return redirect('complex_numbers')
