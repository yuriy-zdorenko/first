Програмування 

	!{--Різні команди--}

	Math.random() (Рандомне число)
	Math.abs() (Повертає абсолютне значення)
	Math.ceil(5.1) (Округлює число у бік зростання)
	Math.floor(5.8) (Округлює число у менший бік)
	Math.round(5.1) (Округлює число за правилами математики)
	.length() (Рахування букв у слові)
	.toUpperCase() (Змінює на верхній регістр)
	.toLowerCase() (Змінює на нижній регістр)
	.replace("beer", "tea") (Заміняє beer на tea)
	.toFixed (В дужках можна обрати числа) (Скорачує рядок до цілого числа)
	.toString() (Перетворює на строчну частину)
	.slice (Виризає щось в рядку)
	.includes ("слово") (Перевірка, чи є такі букви у слові(Приклад: ово - true, аіф - False))
	.indexOf("слово") (Шукає індекс букви чи слова) {Тобто с - 0, в - 3}
	.StartsWith("слово") (Перевіряє початок слова)
	.endsWith("слово") (Перевіряє кінець слова)
	.split ("/") (Розбиває рядок на кілька рядків)
	.join ("?") (Об'єднує рядок через указаний символ)
	.repeat(5) (Повторення рядка)
	.trim() (Видаляє пробіли) {Також є .trimStart() (Видаляє пробіли на початку), .trimEnd() (Видаляє пробіли у кінці)}
	.padStart(5,"Чим добивати") (Скільки по довжині має бути текст) {Наприклад, "слово".padStart(7) = "  слово"}
	.padEnd (5,"Чим добивати") (Працює як padStart)
	parseInt (Перетворює рядок в число)
